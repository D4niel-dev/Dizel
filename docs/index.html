# Local DataFrame analysis with DeepSeek-Coder
import dizel
import pandas as pd

client = dizel.Client(workspace="data-analysis")
df = pd.read_csv("financial_q3.csv")

# Guaranteed privacy, zero cloud transmission
response = client.generate(
    model="deepseek-coder-v2",
    prompt=f"Find anomalies in this dataframe: {df.head()}"
)
print(response.text)
// Low-latency local generation in Rust
use dizel_rs::Client;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Client::new("local-env");
    
    let completion = client
        .model("llama-3-8b")
        .prompt("Sanitize this backend payload...")
        .stream()
        .await?;

    println!("{:?}", completion);
    Ok(())
}
